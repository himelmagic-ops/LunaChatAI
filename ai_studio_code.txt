import React, { useState, useEffect, useRef } from 'react';
import { 
  MessageSquare, 
  Users, 
  Cpu, 
  Send, 
  Plus, 
  Settings, 
  Sparkles, 
  History, 
  Zap, 
  ShieldCheck,
  ChevronRight,
  Database
} from 'lucide-react';

// --- INITIAL MOCK DATA ---
const INITIAL_CHARACTERS = [
  { id: 1, name: "Luna", role: "AI Assistant", avatar: "🌙", description: "The core intelligence of LunaChatAI.", modelId: "luna-base-v1" },
  { id: 2, name: "Aria", role: "Creative Writer", avatar: "🎨", description: "Specializes in storytelling and poetry.", modelId: "luna-base-v1" },
  { id: 3, name: "Kael", role: "Code Expert", avatar: "💻", description: "Technical specialist for debugging and logic.", modelId: "luna-base-v1" }
];

const INITIAL_MODELS = [
  { id: "luna-base-v1", name: "Luna Base v1", version: "1.0.0", type: "General Purpose", base: "Root", temperature: 0.7 },
  { id: "luna-logic-alpha", name: "Luna Logic Alpha", version: "0.9.4", type: "Reasoning", base: "Luna Base v1", temperature: 0.2 }
];

const LunaChatAI = () => {
  // --- STATE ---
  const [view, setView] = useState('chat'); // chat, gallery, models
  const [characters, setCharacters] = useState(INITIAL_CHARACTERS);
  const [models, setModels] = useState(INITIAL_MODELS);
  const [selectedChar, setSelectedChar] = useState(INITIAL_CHARACTERS[0]);
  const [chatHistory, setChatHistory] = useState({}); // { charId: [messages] }
  const [input, setInput] = useState("");
  const [isTraining, setIsTraining] = useState(false);
  const [newModelData, setNewModelData] = useState({ name: '', base: 'luna-base-v1', temp: 0.7 });

  const scrollRef = useRef(null);

  // --- ACTIONS ---
  const handleSendMessage = () => {
    if (!input.trim()) return;

    const userMessage = { role: 'user', content: input, timestamp: new Date().toLocaleTimeString() };
    const charMessages = chatHistory[selectedChar.id] || [];
    
    const updatedMessages = [...charMessages, userMessage];
    setChatHistory({ ...chatHistory, [selectedChar.id]: updatedMessages });
    setInput("");

    // Simulate AI Response based on assigned model
    setTimeout(() => {
      const currentModel = models.find(m => m.id === selectedChar.modelId) || models[0];
      const aiResponse = { 
        role: 'assistant', 
        content: `[Response generated by ${currentModel.name}] This is a demonstration of LunaChatAI's unlimited interaction system. How can I assist you further?`, 
        timestamp: new Date().toLocaleTimeString() 
      };
      setChatHistory(prev => ({
        ...prev,
        [selectedChar.id]: [...(prev[selectedChar.id] || []), aiResponse]
      }));
    }, 800);
  };

  const createNewModel = (e) => {
    e.preventDefault();
    const modelId = newModelData.name.toLowerCase().replace(/\s+/g, '-');
    const newModel = {
      id: modelId,
      name: newModelData.name,
      version: "1.0.0 (Custom)",
      type: "User Trained",
      base: newModelData.base,
      temperature: newModelData.temp
    };
    setModels([...models, newModel]);
    setIsTraining(false);
    setNewModelData({ name: '', base: 'luna-base-v1', temp: 0.7 });
  };

  // --- UI COMPONENTS ---
  const SidebarItem = ({ icon: Icon, label, active, onClick }) => (
    <button 
      onClick={onClick}
      className={`w-full flex items-center space-x-3 px-4 py-3 rounded-xl transition-all ${
        active ? 'bg-indigo-600 text-white shadow-lg shadow-indigo-500/30' : 'text-slate-400 hover:bg-slate-800 hover:text-white'
      }`}
    >
      <Icon size={20} />
      <span className="font-medium">{label}</span>
    </button>
  );

  return (
    <div className="flex h-screen bg-slate-950 text-slate-100 font-sans">
      
      {/* LEFT SIDEBAR */}
      <aside className="w-64 border-r border-slate-800 flex flex-col p-4 space-y-8 bg-slate-900/50 backdrop-blur-xl">
        <div className="flex items-center space-x-3 px-2">
          <div className="w-10 h-10 bg-indigo-600 rounded-lg flex items-center justify-center">
            <Sparkles className="text-white" />
          </div>
          <h1 className="text-xl font-bold tracking-tight bg-gradient-to-r from-white to-indigo-400 bg-clip-text text-transparent">
            LunaChatAI
          </h1>
        </div>

        <nav className="flex-1 space-y-2">
          <SidebarItem icon={MessageSquare} label="Chat" active={view === 'chat'} onClick={() => setView('chat')} />
          <SidebarItem icon={Users} label="Characters" active={view === 'gallery'} onClick={() => setView('gallery')} />
          <SidebarItem icon={Cpu} label="Model Center" active={view === 'models'} onClick={() => setView('models')} />
        </nav>

        <div className="p-4 bg-slate-800/50 rounded-2xl border border-slate-700/50">
          <div className="flex items-center justify-between mb-2">
            <span className="text-xs font-semibold text-slate-400 uppercase tracking-wider">Access Status</span>
            <ShieldCheck size={14} className="text-emerald-400" />
          </div>
          <p className="text-sm text-emerald-400 font-medium">Unlimited Premium</p>
        </div>
      </aside>

      {/* MAIN CONTENT AREA */}
      <main className="flex-1 flex flex-col relative overflow-hidden">
        
        {/* VIEW: CHAT INTERFACE */}
        {view === 'chat' && (
          <div className="flex flex-1 overflow-hidden">
            <div className="flex-1 flex flex-col">
              {/* Chat Header */}
              <header className="h-16 border-b border-slate-800 flex items-center justify-between px-6 bg-slate-950/50">
                <div className="flex items-center space-x-3">
                  <span className="text-2xl">{selectedChar.avatar}</span>
                  <div>
                    <h2 className="font-bold">{selectedChar.name}</h2>
                    <p className="text-xs text-slate-500">Model: {models.find(m => m.id === selectedChar.modelId)?.name}</p>
                  </div>
                </div>
                <div className="flex items-center space-x-4">
                  <button className="p-2 text-slate-400 hover:text-white"><History size={20}/></button>
                  <button className="p-2 text-slate-400 hover:text-white"><Settings size={20}/></button>
                </div>
              </header>

              {/* Chat Messages */}
              <div className="flex-1 overflow-y-auto p-6 space-y-6">
                {(chatHistory[selectedChar.id] || []).length === 0 && (
                  <div className="h-full flex flex-col items-center justify-center text-slate-500 space-y-4">
                    <div className="w-16 h-16 bg-slate-900 rounded-full flex items-center justify-center border border-slate-800">
                      <MessageSquare size={32} />
                    </div>
                    <p>Start a conversation with {selectedChar.name}</p>
                  </div>
                )}
                {(chatHistory[selectedChar.id] || []).map((msg, i) => (
                  <div key={i} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                    <div className={`max-w-[70%] rounded-2xl px-4 py-3 ${
                      msg.role === 'user' ? 'bg-indigo-600 text-white' : 'bg-slate-800 text-slate-200 border border-slate-700'
                    }`}>
                      <p className="text-sm">{msg.content}</p>
                      <span className="text-[10px] opacity-50 mt-1 block">{msg.timestamp}</span>
                    </div>
                  </div>
                ))}
                <div ref={scrollRef} />
              </div>

              {/* Chat Input */}
              <div className="p-4 bg-slate-950 border-t border-slate-800">
                <div className="max-w-4xl mx-auto relative">
                  <input 
                    value={input}
                    onChange={(e) => setInput(e.target.value)}
                    onKeyPress={(e) => e.key === 'Enter' && handleSendMessage()}
                    placeholder={`Message ${selectedChar.name}...`}
                    className="w-full bg-slate-900 border border-slate-700 rounded-xl py-4 pl-6 pr-14 focus:outline-none focus:ring-2 focus:ring-indigo-500 transition-all"
                  />
                  <button 
                    onClick={handleSendMessage}
                    className="absolute right-3 top-1/2 -translate-y-1/2 p-2 bg-indigo-600 rounded-lg hover:bg-indigo-500 transition-colors"
                  >
                    <Send size={18} />
                  </button>
                </div>
              </div>
            </div>

            {/* Right Chat Sidebar */}
            <div className="w-72 border-l border-slate-800 p-6 space-y-6 hidden lg:block">
              <h3 className="text-sm font-semibold text-slate-400 uppercase tracking-widest">Active Character</h3>
              <div className="text-center space-y-4">
                <div className="w-24 h-24 bg-indigo-900/30 rounded-full mx-auto flex items-center justify-center text-5xl border border-indigo-500/30 shadow-2xl shadow-indigo-500/10">
                  {selectedChar.avatar}
                </div>
                <div>
                  <h4 className="text-lg font-bold">{selectedChar.name}</h4>
                  <p className="text-sm text-indigo-400">{selectedChar.role}</p>
                </div>
                <p className="text-xs text-slate-400 leading-relaxed italic">
                  "{selectedChar.description}"
                </p>
              </div>
              <div className="pt-6 border-t border-slate-800">
                <h3 className="text-sm font-semibold text-slate-400 uppercase tracking-widest mb-4">Core Model</h3>
                <div className="bg-slate-900 p-3 rounded-lg border border-slate-800 flex items-center space-x-3">
                  <Database size={16} className="text-indigo-400" />
                  <span className="text-xs truncate">{models.find(m => m.id === selectedChar.modelId)?.name}</span>
                </div>
              </div>
            </div>
          </div>
        )}

        {/* VIEW: CHARACTER GALLERY */}
        {view === 'gallery' && (
          <div className="p-8 overflow-y-auto">
            <div className="flex justify-between items-end mb-8">
              <div>
                <h2 className="text-3xl font-bold mb-2">Character Gallery</h2>
                <p className="text-slate-400">Discover and interact with various AI personalities.</p>
              </div>
              <button className="flex items-center space-x-2 bg-indigo-600 px-4 py-2 rounded-lg hover:bg-indigo-500 transition-colors">
                <Plus size={18} />
                <span>Create Character</span>
              </button>
            </div>
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
              {characters.map(char => (
                <div 
                  key={char.id}
                  onClick={() => { setSelectedChar(char); setView('chat'); }}
                  className="group bg-slate-900 border border-slate-800 p-6 rounded-2xl hover:border-indigo-500 transition-all cursor-pointer hover:shadow-2xl hover:shadow-indigo-500/5"
                >
                  <div className="flex justify-between items-start mb-4">
                    <span className="text-4xl">{char.avatar}</span>
                    <div className="px-2 py-1 bg-slate-800 rounded text-[10px] text-slate-400 font-bold uppercase tracking-tighter">
                      {char.modelId}
                    </div>
                  </div>
                  <h3 className="text-xl font-bold mb-1 group-hover:text-indigo-400 transition-colors">{char.name}</h3>
                  <p className="text-indigo-400 text-sm mb-3">{char.role}</p>
                  <p className="text-slate-400 text-sm line-clamp-2">{char.description}</p>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* VIEW: MODEL TRAINING CENTER */}
        {view === 'models' && (
          <div className="p-8 overflow-y-auto">
            <div className="flex justify-between items-end mb-8">
              <div>
                <h2 className="text-3xl font-bold mb-2">Model Training Center</h2>
                <p className="text-slate-400">Select, adjust, and deploy custom neural models for your characters.</p>
              </div>
              <button 
                onClick={() => setIsTraining(true)}
                className="flex items-center space-x-2 bg-indigo-600 px-4 py-2 rounded-lg hover:bg-indigo-500 transition-colors"
              >
                <Plus size={18} />
                <span>Add New Model</span>
              </button>
            </div>

            {/* Model List Table */}
            <div className="bg-slate-900 border border-slate-800 rounded-2xl overflow-hidden">
              <table className="w-full text-left">
                <thead className="bg-slate-800/50 text-slate-400 text-xs uppercase tracking-widest">
                  <tr>
                    <th className="px-6 py-4">Model Name</th>
                    <th className="px-6 py-4">Base Architecture</th>
                    <th className="px-6 py-4">Type</th>
                    <th className="px-6 py-4">Config</th>
                    <th className="px-6 py-4">Status</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-slate-800">
                  {models.map(model => (
                    <tr key={model.id} className="hover:bg-slate-800/30 transition-colors">
                      <td className="px-6 py-4">
                        <div className="flex items-center space-x-3">
                          <div className="p-2 bg-indigo-500/10 rounded-lg text-indigo-400"><Zap size={16} /></div>
                          <div>
                            <div className="font-bold">{model.name}</div>
                            <div className="text-[10px] text-slate-500">{model.version}</div>
                          </div>
                        </div>
                      </td>
                      <td className="px-6 py-4 text-sm text-slate-400">{model.base}</td>
                      <td className="px-6 py-4">
                        <span className="text-xs px-2 py-1 bg-slate-800 rounded-full border border-slate-700">{model.type}</span>
                      </td>
                      <td className="px-6 py-4 text-xs font-mono text-indigo-400">TEMP: {model.temperature}</td>
                      <td className="px-6 py-4">
                        <div className="flex items-center text-emerald-400 text-xs font-bold">
                          <div className="w-2 h-2 bg-emerald-400 rounded-full mr-2 animate-pulse" />
                          DEPLOYED
                        </div>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>

            {/* Training Modal */}
            {isTraining && (
              <div className="fixed inset-0 bg-black/80 backdrop-blur-sm flex items-center justify-center z-50 p-4">
                <div className="bg-slate-900 border border-slate-700 w-full max-w-md rounded-2xl p-8 shadow-2xl">
                  <h3 className="text-2xl font-bold mb-6 flex items-center space-x-2">
                    <Cpu className="text-indigo-500" />
                    <span>Train New Model</span>
                  </h3>
                  <form onSubmit={createNewModel} className="space-y-6">
                    <div>
                      <label className="block text-sm font-semibold mb-2 text-slate-400">Model Name</label>
                      <input 
                        required
                        value={newModelData.name}
                        onChange={e => setNewModelData({...newModelData, name: e.target.value})}
                        className="w-full bg-slate-800 border border-slate-700 rounded-lg p-3 focus:outline-none focus:ring-2 focus:ring-indigo-500"
                        placeholder="e.g. Luna-Creative-v2"
                      />
                    </div>
                    <div>
                      <label className="block text-sm font-semibold mb-2 text-slate-400">Base Model Architecture</label>
                      <select 
                        value={newModelData.base}
                        onChange={e => setNewModelData({...newModelData, base: e.target.value})}
                        className="w-full bg-slate-800 border border-slate-700 rounded-lg p-3 focus:outline-none"
                      >
                        {models.map(m => <option key={m.id} value={m.id}>{m.name}</option>)}
                      </select>
                    </div>
                    <div>
                      <div className="flex justify-between mb-2">
                        <label className="block text-sm font-semibold text-slate-400">Temperature (Creativity)</label>
                        <span className="text-indigo-400 font-mono">{newModelData.temp}</span>
                      </div>
                      <input 
                        type="range" min="0" max="1" step="0.1"
                        value={newModelData.temp}
                        onChange={e => setNewModelData({...newModelData, temp: parseFloat(e.target.value)})}
                        className="w-full accent-indigo-500"
                      />
                    </div>
                    <div className="flex space-x-3 pt-4">
                      <button 
                        type="button" 
                        onClick={() => setIsTraining(false)}
                        className="flex-1 px-4 py-3 bg-slate-800 rounded-xl font-bold hover:bg-slate-700"
                      >
                        Cancel
                      </button>
                      <button 
                        type="submit"
                        className="flex-1 px-4 py-3 bg-indigo-600 rounded-xl font-bold hover:bg-indigo-500 shadow-lg shadow-indigo-500/20"
                      >
                        Begin Training
                      </button>
                    </div>
                  </form>
                </div>
              </div>
            )}
          </div>
        )}
      </main>
    </div>
  );
};

export default LunaChatAI;
